Integrasi dengan Google Colab

Untuk menggunakan model machine learning dari Google Colab, salin kode berikut dan tempelkan di notebook Colab Anda:

# Kode untuk integrasi dengan Google Colab
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
import pickle

# Load model (ganti dengan path model Anda)
with open('car_price_model.pkl', 'rb') as file:
    model = pickle.load(file)

# Fungsi untuk prediksi harga
def predict_car_price(brand, model_year, mileage, transmission, fuel, engine_size, condition):
    # Preprocess input
    input_data = pd.DataFrame({
        'brand': [brand],
        'year': [model_year],
        'mileage': [mileage],
        'transmission': [transmission],
        'fuel': [fuel],
        'engine_size': [engine_size],
        'condition': [condition]
    })
    
    # Lakukan prediksi
    prediction = model.predict(input_data)
    return prediction[0]
